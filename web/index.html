function humanDate($date) { 
    // comparaison avec aujourd'hui
    $dateDiff = $date->diff(new \DateTime('00:00'));
    if (!$dateDiff->invert && $dateDiff->d == 0) {
        // hier
        $output = 'Hier à ' . $date->format('H:i');
    } else {
        // pas hier
        $output = $date->format('\L\e d/m \à H:i');
    }
}

$articleDate = new \DateTime('yesterday 16:08');
echo humanDate($articleDate);
